/**
 * Render computed channels - one chart per unit group
 * Orchestrates: data preparation â†’ chart creation â†’ interactions â†’ labels
 * 
 * âœ… REFACTORED: Now accepts cfg + data like renderAnalogCharts/renderDigitalCharts
 * 
 * Structure:
 * - cfg.computedChannels[i] = { id, name, unit, equation, color, group, ... } (METADATA)
 * - data.computedData[i] = [value1, value2, ...] (VALUES ARRAY)
 * 
 * @param {Object} cfg - Configuration with computedChannels metadata array
 * @param {Object} data - Data object with computedData values array and time array
 * @param {HTMLElement} chartsContainer - Container for chart elements
 * @param {Array} charts - Array storing all chart instances
 * @param {Array} verticalLinesX - Reactive array of vertical line positions
 * @param {Object} channelState - Reactive state object
 */
export function renderComputedChannels(
  cfg,
  data,
  chartsContainer,
  charts,
  verticalLinesX,
  channelState
) {
  console.log(`[renderComputedChannels] ðŸŸª Starting: ${cfg?.computedChannels?.length || 0} computed channels`);

  // Build combined channel objects from separate cfg + data
  // This creates objects compatible with existing helper functions
  const allComputedChannels = buildComputedChannelsFromCfgAndData(cfg, data);

  // âœ… DEBUG: Log all computed channels and their groups BEFORE filtering
  console.log(`[renderComputedChannels] ðŸ” DEBUG - All computed channels BEFORE ownership filter:`);
  allComputedChannels.forEach((ch, i) => {
    console.log(`  [${i}] id="${ch.id}", group="${ch.group}"`);
  });

  // âœ… OWNERSHIP RULE: Filter computed channels
  // Rule: If a computed channel's group matches an analog group, it's OWNED by that analog chart
  // Therefore, exclude it from standalone computed charts
  // This ensures mutual exclusivity: each channel renders in exactly ONE place
  const analogGroupIds = channelState?.analog?.groups || [];
  
  // âœ… DEBUG: Log analog groups as unique set
  const uniqueAnalogGroups = [...new Set(analogGroupIds.filter(g => g))];
  console.log(
    `[renderComputedChannels] ðŸ“‹ Ownership rule: Unique analog groups are [${uniqueAnalogGroups.join(", ")}]`
  );

  const computedChannels = filterUnassignedComputedChannels(
    allComputedChannels,
    analogGroupIds
  );

  // âœ… DEBUG: Log channels AFTER filtering
  console.log(`[renderComputedChannels] ðŸ” DEBUG - Computed channels AFTER ownership filter: ${computedChannels.length} channels`);
  computedChannels.forEach((ch, i) => {
    console.log(`  [${i}] id="${ch.id}", group="${ch.group}" (will render in standalone)`);
  });

  if (computedChannels.length === 0) {
    console.log(`[renderComputedChannels] âœ… No unassigned computed channels - all owned by analog groups`);
    return;
  }

  // Group by unit and render
  const unitGroups = groupChannelsByUnit(computedChannels);

  // Clean old computed charts
  cleanupOldComputedCharts(charts, chartsContainer);

  // Resolve time array
  const timeArray = resolveTimeArray(data, computedChannels);

  // Get global axis alignment
  const maxYAxes = getMaxYAxes() || 1;

  // Render one chart per unit group
  for (const [unitKey, unitChannels] of unitGroups.entries()) {
    processComputedUnitGroup(
      unitKey,
      unitChannels,
      timeArray,
      maxYAxes,
      chartsContainer,
      charts,
      verticalLinesX,
      channelState,
      data
    );
  }

  console.log(`[renderComputedChannels] âœ… Created ${unitGroups.size} chart(s) with ${computedChannels.length} series`);
}

/**
 * Build combined channel objects from separate cfg and data
 * âœ… NEW: Combines cfg.computedChannels (metadata) with data.computedData (values)
 * 
 * @param {Object} cfg - Configuration with computedChannels array
 * @param {Object} data - Data object with computedData array
 * @returns {Array} Combined channel objects with both metadata and data
 */
function buildComputedChannelsFromCfgAndData(cfg, data) {
  const cfgChannels = cfg?.computedChannels || [];
  const dataValues = data?.computedData || [];
  
  // âœ… ALWAYS use cfg.computedChannels for metadata (source of truth for group, color, etc.)
  // This ensures group changes in cfg are always reflected, even if data has legacy format
  
  // âœ… DEBUG: Detect which format data is in
  const isLegacyDataFormat = dataValues.length > 0 && !Array.isArray(dataValues[0]) && typeof dataValues[0] === 'object';
  console.log(`[buildComputedChannelsFromCfgAndData] ðŸ” Format detection:`, {
    cfgChannelsCount: cfgChannels.length,
    dataValuesCount: dataValues.length,
    isLegacyDataFormat,
    firstDataValueType: dataValues.length > 0 ? (Array.isArray(dataValues[0]) ? 'array' : typeof dataValues[0]) : 'N/A',
  });
  
  // âœ… CRITICAL: Always build from cfg.computedChannels (metadata source of truth)
  // Extract values from either new format (arrays) or legacy format (objects with .data)
  return cfgChannels.map((channelMeta, index) => {
    let values = [];
    
    if (index < dataValues.length) {
      if (Array.isArray(dataValues[index])) {
        // New format: data.computedData[i] is already a values array
        values = dataValues[index];
      } else if (dataValues[index]?.data && Array.isArray(dataValues[index].data)) {
        // Legacy format: extract .data from object
        values = dataValues[index].data;
      } else if (dataValues[index]?.results && Array.isArray(dataValues[index].results)) {
        // Legacy format alternative: extract .results
        values = dataValues[index].results;
      }
    }
    
    console.log(`  [${index}] Using cfg metadata (id="${channelMeta.id}", group="${channelMeta.group}") with ${values.length} values`);
    
    return {
      ...channelMeta,  // â† ALWAYS use cfg for metadata (group, color, etc.)
      data: values,     // Attach values for chart rendering
      results: values,  // Alias for compatibility
    };
  });
}

/**
 * Process a single unit group: create chart, attach interactions, render labels
 * @param {string} unitKey - Unit identifier (e.g., "kV", "A", "unknown")
 * @param {Array} unitChannels - Channels in this unit
 * @param {Array} timeArray - Time array for chart
 * @param {number} maxYAxes - Max Y axes for alignment
 * @param {HTMLElement} chartsContainer - Container
 * @param {Array} charts - Charts array
 * @param {Array} verticalLinesX - Vertical lines state
 * @param {Object} channelState - Channel state
 * @param {Object} data - COMTRADE data object
 */
function processComputedUnitGroup(
  unitKey,
  unitChannels,
  timeArray,
  maxYAxes,
  chartsContainer,
  charts,
  verticalLinesX,
  channelState,
  data
) {
  // Extract metadata
  const groupYLabels = unitChannels.map((ch) => ch.id || "Computed");
  const groupLineColors = unitChannels.map((ch) =>
    ch.color && typeof ch.color === "string" ? ch.color.trim() : ""
  );
  const groupYUnits = unitChannels.map((ch) => ch.unit || unitKey);

  // Create metadata
  const metadata = createComputedChartMetadata({
    unitKey,
    unitChannels,
    lineColors: groupLineColors
  });

  // Build chart data
  const chartData = buildUnitChartData(timeArray, unitChannels);

  // Create drag bar & container
  const dragBar = createDragBar(
    {
      indices: Array.from({ length: unitChannels.length }, (_, i) => i),
      name: `Computed (${unitKey})`,
    },
    {},
    channelState
  );

  const { parentDiv, chartDiv } = attachComputedChartContainer({
    dragBar,
    unitKey,
    groupYLabels,
    groupLineColors,
    metadata,
    chartsContainer
  });

  // Build chart options
  const opts = createChartOptions({
    title: `Computed (${unitKey})`,
    yLabels: groupYLabels,
    lineColors: groupLineColors,
    verticalLinesX,
    xLabel: data.xLabel || "Time",
    xUnit: data.xUnit || "s",
    getCharts: () => charts,
    yUnits: groupYUnits,
    axesScales: [1, ...unitChannels.map(() => 1)],
    singleYAxis: false,
    maxYAxes: maxYAxes,
  });

  // Attach plugins
  attachComputedChartPlugins(opts, verticalLinesX, charts);

  // Create chart
  const chart = initUPlotChart(opts, chartData, chartDiv, charts);

  // Tag chart as computed for later identification
  chart._computed = true;
  chart._computedIds = unitChannels.map((ch) => ch.id);
  chart._type = "computed";
  chart._metadata = metadata;
  chart._userGroupId = metadata.userGroupId;
  chart._uPlotInstance = metadata.uPlotInstance;
  chart._chartType = "computed";
  chart._axesScales = [1, ...unitChannels.map(() => 1)];
  chart._yUnits = groupYUnits || [];
  chart._seriesColors = groupLineColors || [];
  chart._unitKey = unitKey;

  // Attach interactions
  attachComputedChartEventHandlers(
    chart,
    charts,
    verticalLinesX,
    groupYLabels,
    groupLineColors
  );

  // Render labels with equations
  const labels = buildComputedChannelLabels(unitChannels, 
    Array.from({ length: unitChannels.length }, (_, i) => i));
  const labelDiv = parentDiv.querySelector(".chart-label");
  if (labelDiv) {
    labelDiv.innerHTML = "";
    const typeSpan = document.createElement("span");
    typeSpan.textContent = `Computed (${unitKey})`;
    typeSpan.style.cssText = `font-size: 0.7rem; font-weight: 700; color: var(--accent-cyan); text-align: center; text-transform: uppercase; letter-spacing: 0.05em; padding-bottom: 4px; border-bottom: 1px solid var(--border-color); width: 100%;`;
    labelDiv.appendChild(typeSpan);

    unitChannels.forEach((channel, idx) => {
      const channelContainer = document.createElement("div");
      channelContainer.style.cssText = `display:flex;flex-direction:column;align-items:center;gap:4px;width:100%;padding:6px 4px;border:1px solid var(--border-color,#e0e0e0);border-radius:3px;background:var(--bg-tertiary,#f9f9f9);color:var(--chart-text,#333)`;
      const nameContainer = document.createElement("div");
      nameContainer.style.cssText = `display:flex;flex-direction:column;align-items:center;gap:4px;width:100%`;
      const colorDot = document.createElement("span");
      colorDot.style.cssText = `width:10px;height:10px;border-radius:50%;background:${groupLineColors[idx % groupLineColors.length]};border:1px solid var(--border-color)`;
      nameContainer.appendChild(colorDot);
      const nameSpan = document.createElement("span");
      nameSpan.textContent = channel.id;
      nameSpan.style.cssText = `font-size:0.7rem;font-weight:500;color:var(--text-secondary);text-align:center;word-break:break-word;line-height:1.1`;
      nameContainer.appendChild(nameSpan);
      channelContainer.appendChild(nameContainer);
      labelDiv.appendChild(channelContainer);
    });

    renderLatex(labelDiv);
  }
}
